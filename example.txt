    // --- دوال إنشاء القائمة الجانبية ديناميكيًا مع حفظ حالة الطي ---
    function buildSidebar(role) {
        const structure = sidebarStructure[role];
        if (!structure) return;
        const container = document.getElementById('sidebarNavContainer');
        container.innerHTML = '';
        // استرجاع تفضيلات الطي من localStorage
        let openGroups = JSON.parse(localStorage.getItem('sidebar_open_groups')) || {};
        
        structure.groups.forEach(group => {
            const groupId = group.name.replace(/\s/g, '_');
            const isOpen = openGroups[groupId] !== undefined ? openGroups[groupId] : group.defaultOpen;
            // إنشاء عنصر المجموعة
            const groupDiv = document.createElement('div');
            groupDiv.className = 'group-item mb-1';
            // زر المجموعة
            const btn = document.createElement('button');
            btn.className = `sidebar-group-btn w-full flex items-center justify-between px-3 py-2.5 rounded-lg text-gray-700 hover:bg-gray-100 transition`;
            btn.innerHTML = `
                <div class="flex items-center gap-3">
                    <i class="${group.icon} w-5"></i>
                    <span class="text-sm font-medium">${group.name}</span>
                </div>
                <i class="fas fa-chevron-down text-xs transition-transform duration-200 ${isOpen ? 'rotate-180' : ''}"></i>
            `;
            const submenu = document.createElement('div');
            submenu.className = `submenu space-y-1 pr-6 mt-1 ${isOpen ? '' : 'hidden'}`;
            // إضافة عناصر المجموعة
            group.items.forEach(item => {
                const link = document.createElement('a');
                link.href = '#';
                link.className = `submenu-item flex items-center gap-3 px-3 py-2 text-gray-600 hover:bg-gray-50 rounded-lg text-sm transition-all`;
                link.innerHTML = `<i class="${item.icon} w-5"></i><span>${item.name}</span>`;
                link.addEventListener('click', (e) => {
                    e.preventDefault();
                    alert(`تم التوجيه إلى وحدة: ${item.name} (محاكاة تفاعلية)\nالنظام يدعم RBAC وعزل الفروع والذكاء الاصطناعي.`);
                });
                submenu.appendChild(link);
            });
            btn.addEventListener('click', () => {
                const icon = btn.querySelector('.fa-chevron-down');
                submenu.classList.toggle('hidden');
                icon.classList.toggle('rotate-180');
                // حفظ الحالة
                openGroups[groupId] = !submenu.classList.contains('hidden');
                localStorage.setItem('sidebar_open_groups', JSON.stringify(openGroups));
            });
            groupDiv.appendChild(btn);
            groupDiv.appendChild(submenu);
            container.appendChild(groupDiv);
        });
    }

    let charts = {};
    function renderChartsForRole(role) {
        if (charts.salesChart) charts.salesChart.destroy();
        if (charts.branchesChart) charts.branchesChart.destroy();
        if (charts.shiftChart) charts.shiftChart.destroy();
        if (role === 'branch_manager') {
            const ctx = document.getElementById('salesChart')?.getContext('2d');
            if (ctx) charts.salesChart = new Chart(ctx, { type: 'line', data: { labels: ['1 مارس','3','5','7','9','11','13'], datasets: [{ label: 'فعلي', data: [620,670,710,680,730,800,850], borderColor: '#10b981' },{ label: 'توقع Prophet', data: [640,690,730,710,760,830,880], borderColor: '#f59e0b', borderDash: [5,5] }] } });
        } else if (role === 'super_admin') {
            const ctx = document.getElementById('branchesChart')?.getContext('2d');
            if (ctx) charts.branchesChart = new Chart(ctx, { type: 'bar', data: { labels: ['صنعاء','عدن','تعز'], datasets: [{ label: 'المبيعات (ألف $)', data: [15.2,9.8,11.5], backgroundColor: '#3b82f6' }] } });
        } else if (role === 'cashier') {
            const ctx = document.getElementById('shiftChart')?.getContext('2d');
            if (ctx) charts.shiftChart = new Chart(ctx, { type: 'line', data: { labels: ['9ص','10ص','11ص','12م','1م','2م','3م'], datasets: [{ label: 'مبيعات الوردية', data: [20,45,60,85,70,65,55], borderColor: '#10b981' }] } });
        }
    }

    function updateDashboardAndSidebar(role) {
        // تحديث القائمة الجانبية
        buildSidebar(role);
        // تحديث المحتوى
        let html = '';
        if (role === 'branch_manager') html = getBranchManagerDashboard();
        else if (role === 'super_admin') html = getSuperAdminDashboard();
        else if (role === 'cashier') html = getCashierDashboard();
        document.getElementById('dashboardContent').innerHTML = html;
        setTimeout(() => renderChartsForRole(role), 50);
        // تحديث الزر العائم
        const fabDiv = document.getElementById('fabContainer');
        if (role !== 'super_admin') {
            fabDiv.innerHTML = `<button id="fabBtn" class="bg-emerald-600 hover:bg-emerald-700 text-white rounded-full p-4 shadow-xl transition-all flex items-center gap-2"><i class="fas fa-cash-register text-xl"></i><span class="hidden md:inline">بيع سريع</span></button>`;
            document.getElementById('fabBtn')?.addEventListener('click', () => alert("فتح واجهة نقطة البيع السريعة (POS) - يدعم المزامنة التلقائية"));
        } else {
            fabDiv.innerHTML = '';
        }
        // تحديث اسم المستخدم والفرع
        const userSpan = document.getElementById('userNameDisplay');
        const branchSpan = document.getElementById('currentBranchDisplay');
        if (role === 'branch_manager') { userSpan.innerText = 'أحمد العسيري (مدير فرع)'; branchSpan.innerText = 'فرع صنعاء'; }
        else if (role === 'super_admin') { userSpan.innerText = 'سليمان الرئيس (مدير النظام)'; branchSpan.innerText = 'جميع الفروع'; }
        else { userSpan.innerText = 'نور الصيدلي (كاشير)'; branchSpan.innerText = 'فرع صنعاء - كاشير'; }
    }

    // الاستماع لتغيير الدور
    const roleSelect = document.getElementById('roleSelector');
    roleSelect.addEventListener('change', (e) => {
        updateDashboardAndSidebar(e.target.value);
    });
    // تهيئة أولية
    updateDashboardAndSidebar('branch_manager');

    // البحث في القائمة الجانبية
    const searchInput = document.getElementById('sidebarSearch');
    searchInput.addEventListener('input', function(e) {
        const term = e.target.value.trim().toLowerCase();
        const allLinks = document.querySelectorAll('#sidebarNavContainer .submenu-item');
        allLinks.forEach(link => {
            const text = link.innerText.toLowerCase();
            if (term === '' || text.includes(term)) link.style.display = 'flex';
            else link.style.display = 'none';
        });
    });

    // زر طي القائمة للجوال
    const toggleBtn = document.getElementById('toggleSidebarMobile');
    const sidebar = document.getElementById('sidebar');
    toggleBtn.addEventListener('click', () => {
        sidebar.classList.toggle('sidebar-mobile-hidden');
        sidebar.classList.toggle('sidebar-mobile-visible');
    });
