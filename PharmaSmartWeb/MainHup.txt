    <script>
        // 1. تعريف مصفوفة الصلاحيات (أي دور يرى أي وحدات)
        // الأرقام هنا تتطابق مع الـ RoleID في قاعدة بياناتك `userroles`
        const rolePermissions = {
            '1': ['sales', 'inventory', 'finance', 'reports', 'hr', 'admin'], // 1: المدير العام (يرى الكل)
            '2': ['sales', 'inventory', 'finance', 'reports', 'hr'],          // 2: مدير الفرع (يرى كل شيء عدا الإدارة المركزية)
            '5': ['finance', 'reports'],                                      // 5: محاسب مالي (مالية وتقارير فقط)
            '3': ['sales', 'inventory'],                                      // 3: صيدلي (مبيعات ومخزون)
            '6': ['inventory'],                                               // 6: أمين مخزن (مخزون ومشتريات فقط)
            '4': ['sales']                                                    // 4: كاشير (نقاط بيع ومبيعات فقط)
        };

        // معلومات المستخدمين لتغيير الشكل الجمالي عند الاختيار
        const userProfiles = {
            '1': { name: 'المدير العام', role: 'Super Admin', initial: 'أ', color: 'bg-blue-500' },
            '2': { name: 'مدير الفرع', role: 'Branch Manager', initial: 'م', color: 'bg-emerald-500' },
            '5': { name: 'علي (محاسب)', role: 'Accountant', initial: 'ع', color: 'bg-amber-500' },
            '3': { name: 'د. أحمد', role: 'Pharmacist', initial: 'أ', color: 'bg-teal-500' },
            '6': { name: 'صالح (مخازن)', role: 'Storekeeper', initial: 'ص', color: 'bg-orange-500' },
            '4': { name: 'محمد (كاشير)', role: 'Cashier', initial: 'م', color: 'bg-indigo-500' }
        };

        // 2. الكود الذي يتنفذ عند تغيير القائمة المنسدلة
        document.getElementById('roleSelector').addEventListener('change', function() {
            const selectedRoleId = this.value;
            const allowedModules = rolePermissions[selectedRoleId];
            const profile = userProfiles[selectedRoleId];
            
            // تحديث بيانات المستخدم في الـ Header
            document.getElementById('userNameDisplay').textContent = profile.name;
            document.getElementById('userRoleDisplay').textContent = profile.role;
            const initialEl = document.getElementById('userInitial');
            initialEl.textContent = profile.initial;
            initialEl.className = `w-8 h-8 rounded-full flex items-center justify-center text-sm font-bold shadow text-white ${profile.color}`;

            // إخفاء وإظهار الوحدات (المربعات)
            const cards = document.querySelectorAll('.module-card');
            let visibleCount = 0;

            cards.forEach(card => {
                const moduleName = card.getAttribute('data-module');
                
                // إعادة تشغيل الأنيميشن
                card.classList.remove('fade-in');
                void card.offsetWidth; // Trigger reflow
                
                if (allowedModules.includes(moduleName)) {
                    card.style.display = 'block';
                    card.classList.add('fade-in');
                    visibleCount++;
                } else {
                    card.style.display = 'none';
                }
            });

            // إظهار رسالة إذا كان لا يملك أي صلاحيات
            const noModulesMsg = document.getElementById('noModulesMessage');
            if(visibleCount === 0) {
                noModulesMsg.classList.remove('hidden');
            } else {
                noModulesMsg.classList.add('hidden');
            }
        });
    </script>
