    <script>
        // --- 1. مصفوفة الصلاحيات والشاشات (Routing & Roles Matrix) ---
        const systemRoles = {
            '1': { // Super Admin
                name: 'المدير العام', role: 'Super Admin', initial: 'أ', color: 'bg-blue-600',
                allowedMenus: ['dashboard', 'pos', 'inventory', 'purchases', 'journal'], // يرى كل شيء
                defaultView: 'dashboard' // الشاشة الافتراضية
            },
            '4': { // Cashier
                name: 'محمد (كاشير)', role: 'Cashier', initial: 'م', color: 'bg-indigo-500',
                allowedMenus: ['pos'], // يرى نقطة البيع فقط
                defaultView: 'pos' // يدخل مباشرة عليها
            },
            '3': { // Pharmacist
                name: 'د. أحمد', role: 'Pharmacist', initial: 'أ', color: 'bg-teal-500',
                allowedMenus: ['pos', 'inventory'], // مبيعات ومخزون
                defaultView: 'pos'
            },
            '5': { // Accountant
                name: 'علي (محاسب)', role: 'Accountant', initial: 'ع', color: 'bg-amber-500',
                allowedMenus: ['dashboard', 'journal', 'purchases'], // مالية ومشتريات
                defaultView: 'journal'
            },
            '6': { // Storekeeper
                name: 'صالح (مخازن)', role: 'Storekeeper', initial: 'ص', color: 'bg-orange-500',
                allowedMenus: ['inventory', 'purchases'], // مخازن فقط
                defaultView: 'inventory'
            }
        };

        // --- 2. نظام التوجيه (Router Logic) ---
        function navigateTo(viewName, menuItem = null) {
            // إخفاء كل الشاشات
            document.querySelectorAll('.app-view').forEach(view => {
                view.classList.add('hidden');
                view.classList.remove('screen-fade');
            });
            
            // إظهار الشاشة المطلوبة مع تشغيل الأنيميشن
            const activeView = document.getElementById(`view-${viewName}`);
            if(activeView) {
                activeView.classList.remove('hidden');
                void activeView.offsetWidth; // Trigger reflow for animation
                activeView.classList.add('screen-fade');
            }

            // تفعيل الزر في القائمة الجانبية (تصميميا)
            document.querySelectorAll('.nav-item').forEach(nav => {
                nav.classList.remove('bg-blue-600', 'text-white', 'shadow-md');
                nav.classList.add('text-slate-300', 'hover:bg-slate-800');
            });
            
            if(menuItem) {
                menuItem.classList.remove('text-slate-300', 'hover:bg-slate-800');
                menuItem.classList.add('bg-blue-600', 'text-white', 'shadow-md');
                // تحديث عنوان الهيدر
                document.getElementById('headerTitle').textContent = menuItem.innerText.trim();
            } else {
                // في حالة التوجيه التلقائي، نبحث عن الزر الخاص به
                const autoNav = document.querySelector(`.nav-item[data-nav="${viewName}"]`);
                if(autoNav) {
                    autoNav.classList.remove('text-slate-300', 'hover:bg-slate-800');
                    autoNav.classList.add('bg-blue-600', 'text-white', 'shadow-md');
                    document.getElementById('headerTitle').textContent = autoNav.innerText.trim();
                }
            }
        }

        // --- 3. تطبيق الصلاحيات (Apply Role Context) ---
        function applyRole(roleId) {
            const roleData = systemRoles[roleId];

            // تحديث واجهة المستخدم العلوية
            document.getElementById('userName').textContent = roleData.name;
            document.getElementById('userRole').textContent = roleData.role;
            const avatar = document.getElementById('userAvatar');
            avatar.textContent = roleData.initial;
            avatar.className = `w-9 h-9 rounded-md flex items-center justify-center text-white font-bold shadow ${roleData.color}`;

            // إخفاء/إظهار القائمة الجانبية
            document.querySelectorAll('.nav-item').forEach(nav => {
                const navTarget = nav.getAttribute('data-nav');
                nav.style.display = roleData.allowedMenus.includes(navTarget) ? 'flex' : 'none';
            });
            
            // إخفاء/إظهار العناوين الجانبية (Group Titles) بناء على وجود أبناء مرئيين
            document.querySelectorAll('[data-group]').forEach(group => {
                const groupName = group.getAttribute('data-group');
                let hasVisibleChild = false;
                
                if(groupName === 'sales' && (roleData.allowedMenus.includes('pos'))) hasVisibleChild = true;
                if(groupName === 'inventory' && (roleData.allowedMenus.includes('inventory') || roleData.allowedMenus.includes('purchases'))) hasVisibleChild = true;
                if(groupName === 'finance' && (roleData.allowedMenus.includes('journal'))) hasVisibleChild = true;

                group.style.display = hasVisibleChild ? 'block' : 'none';
            });

            // الانتقال التلقائي للشاشة الافتراضية الخاصة بهذا الدور
            navigateTo(roleData.defaultView);
        }

        // --- 4. إعداد الأحداث (Event Listeners) ---
        document.addEventListener('DOMContentLoaded', () => {
            // رسم الشارت للوحة التحكم
            const ctx = document.getElementById('dashChart').getContext('2d');
            new Chart(ctx, {
                type: 'line',
                data: {
                    labels: ['السبت', 'الأحد', 'الاثنين', 'الثلاثاء', 'الأربعاء', 'الخميس'],
                    datasets: [{
                        label: 'المبيعات', data: [120, 190, 150, 250, 220, 300],
                        borderColor: '#1d4ed8', backgroundColor: 'rgba(29, 78, 216, 0.1)',
                        borderWidth: 3, tension: 0.4, fill: true
                    }]
                },
                options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { display: false } } }
            });

            // إعداد أزرار القائمة الجانبية (SPA Routing)
            document.querySelectorAll('.nav-item').forEach(item => {
                item.addEventListener('click', function(e) {
                    e.preventDefault();
                    const viewToLoad = this.getAttribute('data-nav');
                    navigateTo(viewToLoad, this);
                });
            });

            // تغيير الصلاحية من المحاكي
            document.getElementById('roleSelector').addEventListener('change', function() {
                applyRole(this.value);
            });

            // تشغيل التطبيق كمدير عام مبدئياً
            applyRole('1');
        });
